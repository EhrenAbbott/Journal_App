rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    function isSignedIn() {
      return request.auth != null
    }

    match /{document=**} {
	  allow write: if isSignedIn();
      allow read: if request.auth != null && request.auth.uid == resource.data.uid;
    }
  }
}
